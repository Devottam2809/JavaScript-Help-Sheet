  74. My Min / Max JavaScript Methods
    75. 
function myArrayMax(arr) {
    let len = arr.length;
    let max = -Infinity;
    while (len--) {
        if (arr[len] > max) {
        max = arr[len];
        }
    }
    return max;
    }
